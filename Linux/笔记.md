## 设置PATH
export PATH=/usr/local/bin:$PATH

## 设置文件/目录权限
sudo chown -R yourusername:yourgroupname /path/to/directory

## 获取yourusername
id -un

## 获取yourgroupname
id -gn

## 查看文件链接
ls -al

cd /etc/supervisor/conf.d/
